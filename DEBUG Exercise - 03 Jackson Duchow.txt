// This pseudocode is intended to display
// employee net pay values. All employees have a standard
// $45 deduction from their checks.
// If an employee does not earn enough to cover the deduction
// an error message is displayed.
// This example is modularized.
start
   Declarations
      string name
      string EOFNAME = ZZZZ
      housekeeping()
   while name <> EOFNAME
      mainLoop()
   endwhile
   finish()
stop

housekeeping()
   name = input("Enter first name or ", EOFNAME, " to quit")
return

mainLoop()
   Declarations
      num hours
      num rate
      num DEDUCTION = 45
      num net
   hours = input("Enter hours worked for ", name)
   rate = input("Enter hourly rate for ", name)
   gross = hours * rate
   net = gross - DEDUCTION
   if net > 0 then
      output "Net pay for ", name, " is ", net
   else
      output "Deductions not covered. Net is 0."
   endif
   name = input("Enter first name or ", EOFNAME, " to quit")
return

finish()
   output "End of job"
return
